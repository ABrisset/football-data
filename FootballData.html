def fetch(resource, subresource = nil, params = {})
    id, filter = parse_params(params)
    path = "/#{@api_version}/#{resource}"
    path += "/#{id}/#{subresource}" if id
    path += "?#{filter}" if filter
    response = get(path)
    JSON.parse(response.body)
end
